$(document).ready(function(){ $('#imageInput').change(function(){ readURL(this); }); }); function readURL(input) { if (input.files && input.files[0]) { var reader = new FileReader(); reader.onload = function(e) { $('#previewImage').attr('src', e.target.result); $('#previewImage').css('display', 'block'); } reader.readAsDataURL(input.files[0]); // Convert file to data URL } } #previewImage { border: 1px solid #ccc; padding: 5px; margin-top: 10px; }
